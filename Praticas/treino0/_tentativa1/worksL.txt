#include <bits/stdc++.h>
#define IOS ios::sync_with_stdio(0); cin.tie(0); cout.tie(0)
 
using namespace std;
 
bool dec (int a, int b) { return a > b; }
 
typedef long long ll;
 
int main () {
    IOS;
    ll n;
    cin >> n;
 
    ll a[n];
 
    for (ll i = 0; i < n; a[i++] *= -1) cin >> a[i];
 
    sort(a, a+n);
 
    ll ans = 0;
    for (ll i = 0; -a[i]-i-1 > 0; i++) ans += -a[i]-i-1;
 
    cout << ans << endl;
 
    return 0;   
}
