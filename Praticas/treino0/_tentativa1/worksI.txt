#include<bits/stdc++.h>
 
 #define f first
 #define s second
 #define pb push_back
 
 
using namespace std;
typedef pair<int,int> pii;
typedef long long ll;
const int INF = 0x3f3f3f3f;
const ll LINF = 0x3f3f3f3f3f3f3f3fll;
 
 
 
 
 
 
 
int main ()
{
    ios::sync_with_stdio(0);
cin.tie(0);
ll n;
cin>>n;
vector<pair<ll, ll> > v;
map<ll,ll> mapa;
bool confere=true;
for(int i=0;i<n;i++)
{
    ll a;
    ll b;
    cin>>a>>b;
if(mapa.find(b)==mapa.end()){mapa[b]=a;}
    v.pb(make_pair(a,b));
}
 
for(int i=0;i<n;i++)
{
if(v[i].first<mapa[v[i].second]){confere=false;}
else{mapa[v[i].second]=v[i].first;}
}
for(int i=0;i<n;i++)
{
mapa[v[i].second]=v[i].first;
}
for(int i=n-1;i>=0;i--)
{
if(v[i].first>mapa[v[i].second]){confere=false;}
else{mapa[v[i].second]=v[i].first;}
}
if(confere==true){cout<<"YES"<<endl;}
else{cout<<"NO"<<endl;}
return 0;
 
 
}
