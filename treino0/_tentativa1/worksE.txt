#include<bits/stdc++.h>
 
 #define f first
 #define s second
 #define pb push_back
 
 
using namespace std;
typedef pair<int,int> pii;
typedef long long ll;
const int INF = 0x3f3f3f3f;
const ll LINF = 0x3f3f3f3f3f3f3f3fll;
 
 
 
 
 
 
 
int main ()
{
    ios::sync_with_stdio(0);
cin.tie(0);
ll n;
cin>>n;
vector<ll>v (n+1,0);
ll menor=LINF;
for(int i=1;i<=n;i++)
{
int a;
cin>>a;
v[i]=v[i-1]+a;
menor=min(menor,v[i]);
}
ll menosum=-1;
ll zero=0;
cout<<max(zero,menosum*menor)<<endl;
 
return 0;
 
 
}
