#include <bits/stdc++.h>
#define IOS ios::sync_with_stdio(0); cin.tie(0); cout.tie(0)
 
using namespace std;
 
typedef long long ll;
 
int main () {
    IOS;
    ll n;
    cin >> n;
 
    ll par = 0, impar = 0;
 
    for (ll i = 1; i <= n; i++) {
        ll a;
        cin >> a;
 
        if (i%2) impar += a;
        else par += a;
    }
 
    ll q;
    cin >> q;
 
    ll ans, s1, s2;
    if (n%2) {
        ans = impar - par;
        s1 = 1, s2 = -1;
    } else {
        ans = par - impar;
        s1 = -1, s2 = 1;
    }
 
    while (q--) {
        ll l, r, v;
        cin >> l >> r >> v;
 
        if ((r-l)%2) cout << ans << endl;
        else cout << (ans += (l%2 ? s1 : s2)*v) << endl; 
    }
    
    return 0;
}
